<script>
    function removeItem(itemId) {
        document.getElementById(itemId).remove();
        // You would also make a call to the backend to update the cart in the database
        updateTotals();
    }

    function updateQuantity(itemId, change) {
        const quantityInput = document.getElementById('quantity-' + itemId);
        let currentQuantity = parseInt(quantityInput.value);
        currentQuantity += change;
        if (currentQuantity < 1) currentQuantity = 1;
        quantityInput.value = currentQuantity;
        
        updateTotals();
    }

    function updateTotals() {
        let totalAmount = 0;
        const cartItems = document.getElementsByClassName('cart-item');
        for (let item of cartItems) {
            const quantity = parseInt(item.querySelector('input').value);
            const price = parseInt(item.querySelector('.price').innerText);
            totalAmount += quantity * price;
        }
        
        document.getElementById('total-mrp').innerText = '‚Çπ' + (totalAmount + 0).toFixed(2);
        document.getElementById('subtotal').innerText = '‚Çπ' + totalAmount.toFixed(2);
        document.getElementById('total-amount').innerText = '‚Çπ' + (totalAmount - 0 - 0).toFixed(2);
    }
</script>
